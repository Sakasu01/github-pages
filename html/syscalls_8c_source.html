206void * _sbrk_r(
207 struct _reent *_s_r,
208 ptrdiff_t nbytes)
209{
210 char *base; /* errno should be set to ENOMEM on error */
211
212 if (!heap_ptr) { /* Initialize if first time through. */
213 heap_ptr = end;
214 }
215 base = heap_ptr; /* Point to end of heap. */
216
217 if (heap_ptr + nbytes > _heap_end)
218 {
219 errno = ENOMEM;
220 return (caddr_t) -1;
221 }
222 heap_ptr += nbytes; /* Increase heap. */
223
224 return base; /* Return pointer to start of new heap area. */
225}
