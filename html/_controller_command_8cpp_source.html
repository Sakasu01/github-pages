88void resetBoard(uint8_t address, const Init &initParam) {
89 CANProtocol_NodeIDTypedef addr;
90 addr.category = CANProtocol_NodeCategory_ControlalerBoard;
91 addr.id = address;
92
93 CANProtocolMasterCommand_watchDog(controllerCAN, &addr);
94
95 CANProtocol_DataTypedef packet;
96 packet.commandCode = static_cast<uint8_t>(CommandCode::setUpController);
97 packet.dataSize = 8;
98 packet.data[0] = static_cast<uint8_t>(initParam.dcMode);
99 for (uint8_t i = 0; i < 7; ++i) {
100 packet.data[1 + i] =
101 (static_cast<uint8_t>(initParam.adioMode[i * 2 + 1]) & 0x0F)
102 << 4
103 | (static_cast<uint8_t>(initParam.adioMode[i * 2])
104 & 0x0F);
105 }
106
107 CANProtocol_sendAndWaitResponse(controllerCAN,
108 CANProtocol_PacketMasterType_Normal, &addr, &packet);
109}
