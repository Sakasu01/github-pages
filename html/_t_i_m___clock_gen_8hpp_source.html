27constexpr std::optional<std::tuple<uint32_t, uint32_t>> calcPrescalerFromFreq(
28 uint32_t baseFrequency,
29 uint32_t targetFrequency,
30 int32_t period = -1) {
31 // ベース周波数よりも高い or 割り切れない
32 if (baseFrequency < targetFrequency || baseFrequency % targetFrequency != 0) {
33 return {};
34 }
35 const uint32_t division = baseFrequency / targetFrequency;
36 if (period < 0) {
37 uint32_t prescaler = 1;
38 while (static_cast<uint16_t>(0xFFFF) < division / prescaler || division % prescaler != 0) {
39 ++prescaler;
40 }
41 return std::make_tuple(prescaler, division / prescaler);
42 }
43
44 uint16_t periodValue = period;
45 if (division < periodValue || division % periodValue != 0) {
46 return {};
47 }
48 return std::make_tuple(division / periodValue, periodValue);
49}
50// ベース周波数よりも高い場合
51static_assert(!calcPrescalerFromFreq(84e6, 100e6), "");
52// 割り切れない場合
53static_assert(!calcPrescalerFromFreq(84e6, 83e3), "");
54// 84MHz -> 40kHz => prescaler:1 period:2100
55static_assert(calcPrescalerFromFreq(84e6, 40e3) == std::tuple<uint32_t, uint32_t>(1, 2100), "");
56// 168MHz -> 40kHz => prescaler:1 period:4200
57static_assert(calcPrescalerFromFreq(168e6, 40e3) == std::tuple<uint32_t, uint32_t>(1, 4200), "");
58// 84MHz -> 1kHz => prescaler:1 period:4200
59static_assert(calcPrescalerFromFreq(84e6, 1e3) == std::tuple<uint32_t, uint32_t>(2, 42e3), "");
60// 84MHz -> 100Hz => prescaler:14 period:60e3
61static_assert(calcPrescalerFromFreq(84e6, 100) == std::tuple<uint32_t, uint32_t>(14, 60e3), "");

63// 84MHz -> 40kHz, period = 20e3 => ピリオドがありえない
64static_assert(!calcPrescalerFromFreq(84e6, 40e3, 20e3), "");
