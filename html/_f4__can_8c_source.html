62ErrorStatus CAN_config(CAN_TypeDef* CANx, CAN_Bitrate bitrate, bool filter, uint16_t filterSID) {
63 // 初期化用テーブル
64 CAN_SettingTypedef CAN_settings[2] = {
65 CAN_SETTING_INITIALIZER(1),
66 CAN_SETTING_INITIALIZER(2),
67 };
68
69 CAN_SettingTypedef CAN_setting;
70 switch ((uint32_t)CANx) {
71 case (uint32_t)CAN1:
72 CAN_setting = CAN_settings[0];
73 break;
74 case (uint32_t)CAN2:
75 CAN_setting = CAN_settings[1];
76 break;
77 default:
78 return ERROR;
79 }
80
81 //クロック供給
82 RCC_APB2PeriphClockCmd(RCC_APB2Periph_SYSCFG, ENABLE);
83 RCC_APB1PeriphClockCmd(CAN_setting.RCC_CAN, ENABLE);
84
85 //GPIO初期化
86 F4_GPIO_config(&CAN_setting.txPin, GPIO_Fast_Speed, GPIO_Mode_AF,
87 GPIO_OType_PP, GPIO_PuPd_UP);
88 F4_GPIO_config(&CAN_setting.rxPin, GPIO_Fast_Speed, GPIO_Mode_AF,
89 GPIO_OType_PP, GPIO_PuPd_UP);
90
91 //AFは先に設定しないといけないようです
92 GPIO_PinAFConfig(CAN_setting.txPin.GPIOx, CAN_setting.txPin.pinSource,
93 CAN_setting.GPIO_AF);
94 GPIO_PinAFConfig(CAN_setting.rxPin.GPIOx, CAN_setting.rxPin.pinSource,
95 CAN_setting.GPIO_AF);
96
97 //CAN初期化
98 CAN_InitTypeDef CAN_initStructure;
99 CAN_StructInit(&CAN_initStructure);
100 //ビットレート 通信に使う時間単位を設定する
101 CAN_initStructure.CAN_Prescaler = bitrate;
102 //CANのモード
103 CAN_initStructure.CAN_Mode = CAN_Mode_Normal;
104 //再同期を行う際のビット幅を時間単位で指定
105 CAN_initStructure.CAN_SJW = CAN_SJW_1tq;
106 //ビットタイミングレジスタの設定
107 CAN_initStructure.CAN_BS1 = CAN_BS1_2tq;
108 CAN_initStructure.CAN_BS2 = CAN_BS2_3tq;
109 //タイムトリガモード(ハードウェアでタイムスタンプを生成する)
110 CAN_initStructure.CAN_TTCM = DISABLE;
111 //エラー時の自動バスオフ設定
112 CAN_initStructure.CAN_ABOM = ENABLE;
113 //自動再起動モードの設定
114 CAN_initStructure.CAN_AWUM = DISABLE;
115 //自動再送信禁止設定
116 CAN_initStructure.CAN_NART = DISABLE;
117 //FIFOのオーバーラン時にメッセージを破棄するか(DISABLEで上書き)
118 CAN_initStructure.CAN_RFLM = DISABLE;
119 //送信バッファにたまったデータの送信順序(ENABLE:FIFO DISABLE:SID順)
120 CAN_initStructure.CAN_TXFP = ENABLE;
121
122 CAN_DeInit(CANx);
123 if (CAN_Init(CANx, &CAN_initStructure) == CAN_InitStatus_Failed)
124 return ERROR;
125 //CANをスリープモードから解除
126 if (CAN_WakeUp(CANx) == CAN_WakeUp_Failed)
127 return ERROR;
128
129 //CANのフィルタ初期化
130 //細かい事はf4のリファレンスp.1080あたりを参照のこと
131 CAN_FilterInitTypeDef CAN_filterInitStructure;
132 //どのフィルタについて設定するか ちなみに0～13まで設定可能
133 CAN_filterInitStructure.CAN_FilterNumber = CAN_setting.filterNumber;
134 //フィルタモード
135 CAN_filterInitStructure.CAN_FilterMode = CAN_FilterMode_IdMask;
136 //フィルタサイズ
137 CAN_filterInitStructure.CAN_FilterScale = CAN_FilterScale_32bit;
138 //マスク 最上位11ビットを比較
139 CAN_filterInitStructure.CAN_FilterMaskIdHigh = filter ? 0xFFE0 :0x0000;
140 CAN_filterInitStructure.CAN_FilterMaskIdLow = 0x0000;
141 //比較値
142 CAN_filterInitStructure.CAN_FilterIdHigh = filterSID << 5;
143 CAN_filterInitStructure.CAN_FilterIdLow = 0x0000;
144 //フィルタ通過後のデータをFIFOのどれに割り振るか
145 CAN_filterInitStructure.CAN_FilterFIFOAssignment = CAN_FilterFIFO0;
146 //フィルタの有効、無効
147 CAN_filterInitStructure.CAN_FilterActivation = ENABLE;
148 CAN_FilterInit(&CAN_filterInitStructure);
149
150 //割り込み設定
151 NVIC_InitTypeDef NVIC_initStructure;
152 NVIC_initStructure.NVIC_IRQChannel = CAN_setting.CANx_IRQn;
153 NVIC_initStructure.NVIC_IRQChannelPreemptionPriority = CAN_setting.preemptionPriority;
154 NVIC_initStructure.NVIC_IRQChannelSubPriority = CAN_setting.subPriority;
155 NVIC_initStructure.NVIC_IRQChannelCmd = ENABLE;
156 NVIC_Init(&NVIC_initStructure);
157
158 //CAN受信割り込み有効化
159 CAN_ITConfig(CANx, CAN_IT_FMP0, ENABLE);
160
161 return SUCCESS;
162}
