501void SystemInit(void)

503 /* FPU settings ------------------------------------------------------------*/
504 #if (__FPU_PRESENT == 1) && (__FPU_USED == 1)
505 SCB->CPACR |= ((3UL << 10*2)|(3UL << 11*2)); /* set CP10 and CP11 Full Access */
506 #endif
507 /* Reset the RCC clock configuration to the default reset state ------------*/
508 /* Set HSION bit */
509 RCC->CR |= (uint32_t)0x00000001;
510
511 /* Reset CFGR register */
512 RCC->CFGR = 0x00000000;
513
514 /* Reset HSEON, CSSON and PLLON bits */
515 RCC->CR &= (uint32_t)0xFEF6FFFF;
516
517 /* Reset PLLCFGR register */
518 RCC->PLLCFGR = 0x24003010;
519
520 /* Reset HSEBYP bit */
521 RCC->CR &= (uint32_t)0xFFFBFFFF;
522
523 /* Disable all interrupts */
524 RCC->CIR = 0x00000000;
525
526#if defined(DATA_IN_ExtSRAM) || defined(DATA_IN_ExtSDRAM)
527 SystemInit_ExtMemCtl();
528#endif /* DATA_IN_ExtSRAM || DATA_IN_ExtSDRAM */

530 /* Configure the System clock source, PLL Multiplier and Divider factors,
531 AHB/APBx prescalers and Flash settings ----------------------------------*/
532 SetSysClock();
533
534 /* Configure the Vector Table location add offset address ------------------*/

536 SCB->VTOR = SRAM_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal SRAM */

578void SystemCoreClockUpdate(void)
579{
580 uint32_t tmp = 0, pllvco = 0, pllp = 2, pllsource = 0, pllm = 2;
581#if defined(STM32F412xG) || defined(STM32F446xx)
582 uint32_t pllr = 2;
583#endif /* STM32F412xG || STM32F446xx */
584 /* Get SYSCLK source -------------------------------------------------------*/
585 tmp = RCC->CFGR & RCC_CFGR_SWS;
586
587 switch (tmp)
588 {
589 case 0x00: /* HSI used as system clock source */
590 SystemCoreClock = HSI_VALUE;
591 break;
592 case 0x04: /* HSE used as system clock source */
593 SystemCoreClock = HSE_VALUE;
594 break;
595 case 0x08: /* PLL P used as system clock source */
596 /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N
597 SYSCLK = PLL_VCO / PLL_P
598 */
599 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
600 pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
601
602#if defined(STM32F40_41xxx) || defined(STM32F427_437xx) || defined(STM32F429_439xx) || defined(STM32F401xx) || defined(STM32F412xG) || defined(STM32F446xx) || defined(STM32F469_479xx)
603 if (pllsource != 0)
604 {
605 /* HSE used as PLL clock source */
606 pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
607 }
608 else
609 {
610 /* HSI used as PLL clock source */
611 pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
612 }
613#elif defined(STM32F410xx) || defined(STM32F411xE)
614#if defined(USE_HSE_BYPASS)
615 if (pllsource != 0)
616 {
617 /* HSE used as PLL clock source */
618 pllvco = (HSE_BYPASS_INPUT_FREQUENCY / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
619 }
620#else
621 if (pllsource == 0)
622 {
623 /* HSI used as PLL clock source */
624 pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
625 }
626#endif /* USE_HSE_BYPASS */
627#endif /* STM32F40_41xxx || STM32F427_437xx || STM32F429_439xx || STM32F401xx || STM32F412xG || STM32F446xx || STM32F469_479xx */
628 pllp = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLP) >>16) + 1 ) *2;
629 SystemCoreClock = pllvco/pllp;
630 break;
631#if defined(STM32F412xG) || defined(STM32F446xx)
632 case 0x0C: /* PLL R used as system clock source */
633 /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N
634 SYSCLK = PLL_VCO / PLL_R
635 */
636 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
637 pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
638 if (pllsource != 0)
639 {
640 /* HSE used as PLL clock source */
641 pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
642 }
643 else
644 {
645 /* HSI used as PLL clock source */
646 pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
647 }
648
649 pllr = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLR) >>28) + 1 ) *2;
650 SystemCoreClock = pllvco/pllr;
651 break;
652#endif /* STM32F412xG || STM32F446xx */
653 default:
654 SystemCoreClock = HSI_VALUE;
655 break;
656 }
657 /* Set Flash wait state */
658 tmp = FLASH->ACR;
659 tmp &= (uint32_t)((uint32_t)~(uint32_t)0x0000000F);
660 tmp |= ((SystemCoreClock - 1) / 30000000);
661 FLASH->ACR = tmp;
662
663 /* Compute HCLK frequency --------------------------------------------------*/
664 /* Get HCLK prescaler */
665 tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];
666 /* HCLK frequency */
667 SystemCoreClock >>= tmp;
668}

681/******************************************************************************/
682/* PLL (clocked by HSE) used as System clock source */
683/******************************************************************************/
684 __IO uint32_t StartUpCounter = 0, HSEStatus = 0;
685
686 /* Enable HSE */
687 RCC->CR |= ((uint32_t)RCC_CR_HSEON);
688
689 /* Wait till HSE is ready and if Time out is reached exit */
690 do
691 {
692 HSEStatus = RCC->CR & RCC_CR_HSERDY;
693 StartUpCounter++;
694 } while((HSEStatus == 0) && (StartUpCounter != HSE_STARTUP_TIMEOUT));
695
696 if ((RCC->CR & RCC_CR_HSERDY) != RESET)
697 {
698 HSEStatus = (uint32_t)0x01;
699 }
700 else
701 {
702 HSEStatus = (uint32_t)0x00;
703 }
704
705 if (HSEStatus == (uint32_t)0x01)
706 {
707 /* Select regulator voltage output Scale 1 mode */
708 RCC->APB1ENR |= RCC_APB1ENR_PWREN;
709 PWR->CR |= PWR_CR_VOS;
710
711 /* HCLK = SYSCLK / x*/
712 RCC->CFGR |= RCC_CFGR_HPRE_DIVx;
713
714 /* PCLK2 = HCLK / x*/
715 RCC->CFGR |= RCC_CFGR_PPRE2_DIVx;
716

718 RCC->CFGR |= RCC_CFGR_PPRE1_DIVx;

720#if defined(STM32F40_41xxx) || defined(STM32F427_437xx) || defined(STM32F429_439xx) || defined(STM32F401xx) || defined(STM32F469_479xx)
721 /* Configure the main PLL */
722 RCC->PLLCFGR = PLL_M | (PLL_N << 6) | (((PLL_P >> 1) -1) << 16) |
723 (RCC_PLLCFGR_PLLSRC_HSE) | (PLL_Q << 24);
724#endif /* STM32F40_41xxx || STM32F401xx || STM32F427_437x || STM32F429_439xx || STM32F469_479xx */

730#endif /* STM32F412xG || STM32F446xx */

732 /* Enable the main PLL */
733 RCC->CR |= RCC_CR_PLLON;

736 while((RCC->CR & RCC_CR_PLLRDY) == 0)

738 }

768#endif /* STM32F40_41xxx || STM32F412xG */

773#endif /* STM32F401xx */
