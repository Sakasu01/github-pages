使用例
// TIM1を50Hz, ピリオドを20'000で初期化 => キャプチャコンペアの数値がパルス幅の[us]と等しくなる
peripheral::TIM<F4::TIMMethod> tim(F4::TimerClockGenerator<F4::tim1, 50, 20000>{});
// OC1を初期化
tim.configOC(peripheral::tim::Ch::_1);
Servo<F4::TIMMethod> servo;
// TIM1 OC1, PA0(ピンは適当)
servo.config(tim.createChannel(peripheral::tim::Ch::_1), F4::PA0);
tim.setStart(true);
servo.setEnable(true);
servo.setPulseWidth(1500);
