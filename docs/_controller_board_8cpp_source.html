42bool ControllerBoard::config(const InitParameter& initParam)
43{
44 if (!checkVersion())
45 {
46 return false;
47 }
48
49 canprotocol::Packet::Payload payload;
50 commandCode.functionCode = static_cast<uint8_t>(FunctionCode::config);
51
52 payload.setSize(8);
53 payload.data[0] = static_cast<std::byte>(initParam.dcMode);
54 for (uint8_t i = 0; i < 7; ++i)
55 {
56 payload.data[1 + i] =
57 static_cast<std::byte>(
58 (static_cast<uint8_t>(initParam.adioMode[i * 2 + 1]) & 0x0F) << 4
59 | (static_cast<uint8_t>(initParam.adioMode[i * 2]) & 0x0F));
60 }
61
62 auto recv = this->client.sendPacketAndWaitResponse(
63 node, commandCode.bitvar, payload, canprotocol::MasterCode::normal);
64
65 return static_cast<bool>(recv.value().data[0]);
66}
