432void SystemInit(void)
433{
434 /* FPU settings ------------------------------------------------------------*/
435 #if (__FPU_PRESENT == 1) && (__FPU_USED == 1)
436 SCB->CPACR |= ((3UL << 10*2)|(3UL << 11*2)); /* set CP10 and CP11 Full Access */
437 #endif
438 /* Reset the RCC clock configuration to the default reset state ------------*/
439 /* Set HSION bit */
440 RCC->CR |= (uint32_t)0x00000001;
441
442 /* Reset CFGR register */
443 RCC->CFGR = 0x00000000;
444
445 /* Reset HSEON, CSSON and PLLON bits */
446 RCC->CR &= (uint32_t)0xFEF6FFFF;
447
448 /* Reset PLLCFGR register */
449 RCC->PLLCFGR = 0x24003010;
450
451 /* Reset HSEBYP bit */
452 RCC->CR &= (uint32_t)0xFFFBFFFF;
453
454 /* Disable all interrupts */
455 RCC->CIR = 0x00000000;
456
457#if defined(DATA_IN_ExtSRAM) || defined(DATA_IN_ExtSDRAM)
458 SystemInit_ExtMemCtl();
459#endif /* DATA_IN_ExtSRAM || DATA_IN_ExtSDRAM */
460
461 /* Configure the System clock source, PLL Multiplier and Divider factors,
462 AHB/APBx prescalers and Flash settings ----------------------------------*/
463 SetSysClock();
464
465 /* Configure the Vector Table location add offset address ------------------*/
466#ifdef VECT_TAB_SRAM
467 SCB->VTOR = SRAM_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal SRAM */
468#else
469 SCB->VTOR = FLASH_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal FLASH */
470#endif
471}

509void SystemCoreClockUpdate(void)
510{
511 uint32_t tmp = 0, pllvco = 0, pllp = 2, pllsource = 0, pllm = 2;
512
513 /* Get SYSCLK source -------------------------------------------------------*/
514 tmp = RCC->CFGR & RCC_CFGR_SWS;
515
516 switch (tmp)
517 {
518 case 0x00: /* HSI used as system clock source */
519 SystemCoreClock = HSI_VALUE;
520 break;
521 case 0x04: /* HSE used as system clock source */
522 SystemCoreClock = HSE_VALUE;
523 break;
524 case 0x08: /* PLL P used as system clock source */
525 /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N
526 SYSCLK = PLL_VCO / PLL_P
527 */
528 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
529 pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
530
531 if (pllsource != 0)
532 {
533 /* HSE used as PLL clock source */
534 pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
535 }
536 else
537 {
538 /* HSI used as PLL clock source */
539 pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
540 }
541
542 pllp = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLP) >>16) + 1 ) *2;
543 SystemCoreClock = pllvco/pllp;
544 break;
545 default:
546 SystemCoreClock = HSI_VALUE;
547 break;
548 }
549 /* Set Flash wait state */
550 tmp = FLASH->ACR;
551 tmp &= (uint32_t)((uint32_t)~(uint32_t)0x0000000F);
552 tmp |= ((SystemCoreClock - 1) / 30000000);
553 FLASH->ACR = tmp;
554
555 /* Compute HCLK frequency --------------------------------------------------*/
556 /* Get HCLK prescaler */
557 tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];
558 /* HCLK frequency */
559 SystemCoreClock >>= tmp;
560}

570static void SetSysClock(void)
571{
572 __IO uint32_t StartUpCounter = 0, HSEStatus = 0;
573
574 /* Enable HSE */
575 RCC->CR |= ((uint32_t)RCC_CR_HSEON);
576
577 /* Wait till HSE is ready and if Time out is reached exit */
578 do
579 {
580 HSEStatus = RCC->CR & RCC_CR_HSERDY;
581 StartUpCounter++;
582 } while((HSEStatus == 0) && (StartUpCounter != HSE_STARTUP_TIMEOUT));
583
584 if ((RCC->CR & RCC_CR_HSERDY) != RESET)
585 {
586 HSEStatus = (uint32_t)0x01;
587 }
588 else
589 {
590 HSEStatus = (uint32_t)0x00;
591 }
592
593 if (HSEStatus == (uint32_t)0x01)
594 {
595 /* Set SystemCoreClock */
596 SystemCoreClock = HSE_VALUE;
597
598#if defined (STM32F40_41xxx)
599 /* STM32F40x RevA have Errata, DeviceID has 0x0411(Same as STM32F2)! */
600 if((DBGMCU_GetREVID() == 0x2000)&&(DBGMCU_GetDEVID() == 0x0411))
601 {
602 /* STM32F40x RevA Chip S**K! */
603 /* Configure Instruction cache, Data cache and wait state */
604 FLASH->ACR = FLASH_ACR_ICEN | FLASH_ACR_DCEN | ((SystemCoreClock - 1) / 30000000);
605 }
606 /* STM32F40x RevZ */
607 else
608 {
609 /* Configure Flash prefetch, Instruction cache, Data cache and wait state */
610 FLASH->ACR = FLASH_ACR_PRFTEN | FLASH_ACR_ICEN | FLASH_ACR_DCEN | ((SystemCoreClock - 1) / 30000000);
611 }
612#else
613 /* Configure Flash prefetch, Instruction cache, Data cache and wait state */
614 FLASH->ACR = FLASH_ACR_PRFTEN | FLASH_ACR_ICEN | FLASH_ACR_DCEN | ((SystemCoreClock - 1) / 30000000);
615#endif /* STM32F40_41xxx */
616
617 /* Set SystemCoreClock */
618 SystemCoreClock >>= AHBPrescTable[(RCC_CFGR_HPRE_DIVx >> 4)];
619
620 /* Select regulator voltage output Scale 1 mode */
621 RCC->APB1ENR |= RCC_APB1ENR_PWREN;
622 PWR->CR |= PWR_CR_VOS;
623
624 /* HCLK = SYSCLK / 1*/
625 RCC->CFGR |= RCC_CFGR_HPRE_DIVx;
626
627 /* PCLK2 = HCLK / 2*/
628 RCC->CFGR |= RCC_CFGR_PPRE2_DIVx;
629
630 /* PCLK1 = HCLK / 4*/
631 RCC->CFGR |= RCC_CFGR_PPRE1_DIVx;
632
633 /* Select the HSE as system clock source */
634 RCC->CFGR &= (uint32_t)((uint32_t)~(RCC_CFGR_SW));
635 RCC->CFGR |= RCC_CFGR_SW_HSE;
636
637 /* Wait till the HSE is used as system clock source */
638 while ((RCC->CFGR & (uint32_t)RCC_CFGR_SWS ) != RCC_CFGR_SWS_HSE)
639 {
640 }
641 }
642 else
643 { /* If HSE fails to start-up, the application will have wrong clock
644 configuration. User can add here some code to deal with this error */
645 SystemCoreClock = HSI_VALUE;
646 }
