80 bool run(const coords::XYT &targetAccel) {
81 Eigen::Vector2f fi = targetAccel.xy / N;
82// torque distribution
83 for(std::size_t i = 0; i < N; i++){
84 float sum_of_sq = motor.wheelDirectionVec[i][0] * motor.wheelDirectionVec[i][0] + motor.wheelDirectionVec[i][1] * motor.wheelDirectionVec[i][1];
85 float yawMoRatio = targetAccel.theta / (N * sum_of_sq);
86 Eigen::Vector2f targetForce;
87 targetForce << motor.wheelDirectionVec[i][0] * yawMoRatio, motor.wheelDirectionVec[i][1] * yawMoRatio;
88 targetForce += fi;
89 applyForce(targetForce, i);
90 }
91 mother::serial << "\n";
92 applyAngle();
93 applyTorque();
94 return true;
95 }

217 void applyForce(const Eigen::Vector2f &target, std::size_t i) {
218
219 float ang = Gouyoku_getRotationAxisPosition(motor.steer.address[i]) * math::PI2;
220
221 mother::serial << ang << " ";
222
223 float c = std::cos(ang);
224 float s = std::sin(ang);
225 //rotation
226 float wx = c * motor.wheelDirectionVec[i][0]
227 - s * motor.wheelDirectionVec[i][1];
228 float wy = s * motor.wheelDirectionVec[i][0]
229 + c * motor.wheelDirectionVec[i][1];
230
231 float arg = std::atan2(wy, wx);
232
233 Eigen::Rotation2Df r {-arg};
234
235 Eigen::Vector2f rel = r * target;
236
237 drivingTorque[i] = rel.x();
238
239 steeringAngle[i] = rel.y() * 0.6f;
240
241 }

249 void offset(){

255 float potentio[N] = { };

257 for (uint8_t i = 0; i < N; ++i) {
258 if (isMoving[i]) {
259 //potentio
260 potentio[i] =
261 static_cast<float>(mother::adcSensor[i].read());
262 diff[i] = motor.offsetValue[i] - potentio[i];
263 mother::serial << potentio[i] << " ";
264 if (std::abs(diff[i]) < 50.f || true) {
265 isMoving[i] = false;
266 Gouyoku_setOutputAngleVelo(motor.steer.address[i], 0.f);
267 Gouyoku_setFollowingElement(motor.steer.address[i],
268 Gouyoku_PositionControl);
269 Gouyoku_setPositionInitialize(motor.steer.address[i]);
270 } else {
271 Gouyoku_setOutputAngleVelo(motor.steer.address[i],
