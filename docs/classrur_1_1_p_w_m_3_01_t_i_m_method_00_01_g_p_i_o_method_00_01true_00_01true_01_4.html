template<class TIMMethod, class GPIOMethod>
class rur::PWM< TIMMethod, GPIOMethod, true, true >

PWMのクラス

使用例
// TIM1を100kHzで初期化
peripheral::TIM<F4::TIMMethod> tim(F4::TimerClockGenerator<F4::tim1, 100000>{});
// OC1を相補モードで初期化
tim.configOC(peripheral::tim::Ch::_1, peripheral::tim::PWMMode::complementary);
PWM<F4::TIMMethod> pwm;
// TIM1 OC1, p PA0, n PA1で初期化(ピンは適当)
pwm.config(tim.createChannel(peripheral::tim::Ch::_1), F4::PA0, F4::PA1);
tim.setStart(true);
pwm.setEnable(true);
pwm.setDuty(0.3f); // 30%
