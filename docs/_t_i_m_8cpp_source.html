18std::optional<IRQn> getIRQn(uintptr_t periphBase, peripheral::tim::InterruptMode mode) {

20 if (periphBase == TIM1_BASE) {

82bool TIMMethod::configOutputCompare(peripheral::tim::Ch ch, peripheral::tim::PWMMode mode, float deadTime) {
83 TIM_OCInitTypeDef settings;
84 TIM_OCStructInit(&settings);
85 settings.TIM_OCMode = TIM_OCMode_PWM1;
86 settings.TIM_OutputState = TIM_OutputState_Disable;
87 settings.TIM_OutputNState = TIM_OutputNState_Disable;
88 settings.TIM_OCPolarity = TIM_OCPolarity_High;
89 settings.TIM_OCNPolarity = TIM_OCNPolarity_High;
90 settings.TIM_OCIdleState = TIM_OCIdleState_Reset;
91 settings.TIM_OCNIdleState = TIM_OCNIdleState_Reset;
92 settings.TIM_Pulse = 0;
93
94 switch (ch) {
95 case peripheral::tim::Ch::_1:
96 TIM_OC1Init(periphInfo.splPtr(), &settings);
97 break;
98 case peripheral::tim::Ch::_2:
99 TIM_OC2Init(periphInfo.splPtr(), &settings);
100 break;
101 case peripheral::tim::Ch::_3:
102 TIM_OC3Init(periphInfo.splPtr(), &settings);
103 break;
104 case peripheral::tim::Ch::_4:
105 TIM_OC4Init(periphInfo.splPtr(), &settings);
106 break;
107 default:
108 return false;
109 }
110
111 if (mode == peripheral::tim::PWMMode::complementary) {
112 TIM_BDTRInitTypeDef settings;
113 settings.TIM_OSSRState = TIM_OSSRState_Enable;
114 settings.TIM_OSSIState = TIM_OSSIState_Enable;
115 settings.TIM_LOCKLevel = TIM_LOCKLevel_OFF;
116 settings.TIM_Break = TIM_Break_Disable;
117 settings.TIM_BreakPolarity = TIM_BreakPolarity_High;
118 settings.TIM_AutomaticOutput = TIM_AutomaticOutput_Enable;
119
120 // sa refarence ja p586
121 uint32_t deadTimeUnit = std::ceil(deadTime * periphInfo.baseFrequency);
122 uint8_t deadTimeRegister = 0x00;
123 if (deadTimeUnit <= 0x7F) {
124 deadTimeRegister = (0x7F & (deadTimeUnit)) | 0x00;
125 } else if (deadTimeUnit <= 0xFE) {
126 deadTimeRegister = (0x3F & (deadTimeUnit / 2 - 0x40)) | 0x80;
127 } else if (deadTimeUnit <= 0x1F8) {
128 deadTimeRegister = (0x1F & (deadTimeUnit / 8 - 0x20)) | 0xC0;
129 } else if (deadTimeUnit <= 0x3F0) {
130 deadTimeRegister = (0x1F & (deadTimeUnit / 16 - 0x20)) | 0xE0;
131 } else {
132 deadTimeRegister = 0xFF;
133 }
134 settings.TIM_DeadTime = deadTimeRegister;
135 TIM_BDTRConfig(periphInfo.splPtr(), &settings);
136 }
137 TIM_CtrlPWMOutputs(periphInfo.splPtr(), ENABLE);
138 return true;
139}

141bool TIMMethod::configIT(peripheral::tim::InterruptMode mode, uint8_t priority, std::function<void()> function) {
142 auto irqn = getIRQn(periphInfo.baseAddr, mode);
143
144 if (!irqn) {
145 return false;
146 }
147
148 interruptMap.insert(std::make_pair(std::make_pair(periphInfo.baseAddr, mode), function));
149 Interrupt::config(*irqn, priority);
150 uint16_t itMode = util::enumToReg(mode, TIM_IT_Update, TIM_IT_CC1, TIM_IT_CC2, TIM_IT_CC3, TIM_IT_CC4);
151
152 TIM_ClearITPendingBit(periphInfo.splPtr(), itMode);
153 TIM_ITConfig(periphInfo.splPtr(), itMode, ENABLE);
154
155 return true;
156}
