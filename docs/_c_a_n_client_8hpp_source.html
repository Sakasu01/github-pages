60 template <class CANMethod>
61 bool config(peripheral::CAN<CANMethod>& can, const NodeIDCode& ownNode)
62 {
63 using IDType = peripheral::can::Message::IDType;
64 using CANMessage = peripheral::can::Message;
65 bool result = true;
66 result &= can.config();
67
68 //マスターコードのフィルタ設定．
69 //マスターなら内容を問わず通す．
70 uint32_t masterFilterMach = static_cast<uint32_t>(MasterCode::master) << 28;
71 can.appendFilter(0, IDType::Extended, 1 << 28, masterFilterMach);
72 //ノーマルなら受信先が自分のメッセージのみ通す．
73 uint32_t normalFilterMask = 1 << 28 | ((0x07 << 4) | 0x0F) << 8;
74 uint32_t normalFilterMach = (static_cast<uint32_t>(MasterCode::normal) << 28)
75 | (((static_cast<uint32_t>(ownNode.category.get()) & 0x07) << 4)
76 | (static_cast<uint32_t>(ownNode.id) & 0x0F))
77 << 8;
78 can.appendFilter(
79 1, IDType::Extended, normalFilterMask, normalFilterMach);
80
81 //CANの割り込み設定．
82 std::function<void(const CANMessage&)> canIT =
83 [&](const CANMessage& msg) { provider.processReceivingData(canprotocol::convertMsgToPacket(msg)); };
84 can.configReceiveMessageIT(0, canIT);
85
86 std::function<void(const Packet&)> masterCallback =
87 [&](const auto& p) { masterClient.runCommand(p); };
88 std::function<void(const Packet&)> sendFunc =
89 [&](const Packet& p) { can.sendMessage(canprotocol::convertPacketToMsg(p)); };
90 result &= provider.config(ownNode, masterCallback, sendFunc);
91
92 masterClient.config(ownNode);
93 masterClient.setProvider(&provider);
94
95 std::function<bool()> resetCallback =
96 [&] {NVIC_SystemReset(); return true; };
97 masterClient.setResetCallBack(resetCallback);
98
99 masterClient.setErrorCallBack(
100 [&](std::uint8_t errorCode, const NodeIDCode& sender) { errorCallBack(static_cast<MasterCommmandErrorCode>(errorCode), sender); });
101
102 masterClient.sendCheckConnected();
103
104 return result;
105 }
