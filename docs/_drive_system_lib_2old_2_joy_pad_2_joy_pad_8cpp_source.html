120void JoyPad_readData(void) {
121 //idが連続で同じであった回数のカウンタ
122 static uint32_t dataNotReadCount = 0;
123
124 SerialProtocol_parseData(&serialPort, &protocol, 1);
125
126 if (protocol.detectCount == 0) {
127 dataNotReadCount++;
128 if (JOYPAD_DISCONNECT_COUNT < dataNotReadCount) {
129 JoyPad_clearData();
130 latestPadData.errorCode = Disconnect;
131 }
132 return;
133 } else {
134 dataNotReadCount = 0;
135 latestPadData.errorCode = NoneError;
136 }
137
138 //チャタリング対策用のやつ
139 static uint16_t oldButtonBit = 0x0000;
140 uint16_t nowButtonBit = JoyPad_struct2Bits(latestPadData.button);
141 latestPadData.button = JoyPad_bits2Struct(nowButtonBit & oldButtonBit);
142 oldButtonBit = nowButtonBit;
143}
