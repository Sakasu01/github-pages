32bool I2CMethod::config(uint32_t clockSpeed, peripheral::Pin sclPin, peripheral::Pin sdaPin) {
33 auto af = i2cAF(periphInfo.baseAddr);
34 if (!af) {
35 return false;
36 }
37 GPIO scl(sclPin), sda(sdaPin);
38 scl.config(GPIO::Mode::alternate, GPIO::Pull::noPull, GPIO::OType::od);
39 sda.config(GPIO::Mode::alternate, GPIO::Pull::noPull, GPIO::OType::od);
40 scl.configAF(*af);
41 sda.configAF(*af);
42
43 periphInfo.enableClock();
44
45 I2C_DeInit(periphInfo.splPtr());
46 I2C_InitTypeDef settings;
47 I2C_StructInit(&settings);
48 settings.I2C_ClockSpeed = clockSpeed;
49 settings.I2C_DutyCycle = I2C_DutyCycle_2;
50 settings.I2C_Mode = I2C_Mode_I2C;
51 settings.I2C_OwnAddress1 = 0;
52 settings.I2C_Ack = I2C_Ack_Enable;
53 settings.I2C_AcknowledgedAddress = I2C_AcknowledgedAddress_7bit;
54 I2C_Init(periphInfo.splPtr(), &settings);
55 I2C_Cmd(periphInfo.splPtr(), ENABLE);
56
57 return true;
58}

80bool I2CMethod::sendRepeatedStartCondition(std::bitset<7> addr, peripheral::i2c::Direction dir, std::size_t dataSize, std::optional<uint32_t> timeoutUsec) {
81 I2C_GenerateSTART(periphInfo.splPtr(), ENABLE);
82 util::ElapsedTimer et;
83 while (!I2C_CheckEvent(periphInfo.splPtr(), I2C_EVENT_MASTER_MODE_SELECT)) {
84 if(timeoutUsec && *timeoutUsec < et.getElapsedUsec()) {
85 return false;
86 }
87 }
88 uint8_t dirReg = util::enumToReg(dir, I2C_Direction_Transmitter, I2C_Direction_Receiver);
89 uint32_t event = util::enumToReg(dir, I2C_EVENT_MASTER_TRANSMITTER_MODE_SELECTED, I2C_EVENT_MASTER_RECEIVER_MODE_SELECTED);
90 I2C_AcknowledgeConfig(periphInfo.splPtr(), dir == peripheral::i2c::Direction::receive ? ENABLE : DISABLE);
91 I2C_Send7bitAddress(periphInfo.splPtr(), addr.to_ulong() << 1, dirReg);
92 et.restart();
93 while (!I2C_CheckEvent(periphInfo.splPtr(), event)) {
94 if(timeoutUsec && *timeoutUsec < et.getElapsedUsec()) {
95 return false;
96 }
97 }
98 return true;
99}
