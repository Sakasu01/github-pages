179template <class Method>
180void UART<Method>::updateRxBufferLen() {
181 const auto s = rxBuffer.size();
182 if (!s) return;
183
184 auto tail = rxDMA.nextTransmitDataPos();
185 // 正規化インデックスでheadを先頭とした座標を取得
186 auto ni = util::normalIndex(tail - rxHead, s);
187 if (rxLen <= ni) { // データがheadを侵食していない場合(前回のtailも浸食していた場合...?知らんな)
188 rxLen = ni;
189 } else { // データがheadを侵食していた場合(ni分のデータは消失)
190 InterruptLocker locker;
191 rxHead = tail;
192 rxLen = s;
193 }
194}
