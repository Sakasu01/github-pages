65 periphInfo.enableClock();
66
67 {

70 settings.ADC_Prescaler = ADC_Prescaler_Div2;
71 settings.ADC_DMAAccessMode = ADC_DMAAccessMode_Disabled;
72 settings.ADC_TwoSamplingDelay = ADC_TwoSamplingDelay_5Cycles;
73 ADC_CommonInit(&settings);
74 }
75
76 {
77 ADC_InitTypeDef settings;
78 ADC_StructInit(&settings);
79 settings.ADC_Resolution = ADC_Resolution_12b;
80 settings.ADC_ScanConvMode = ENABLE;
81 settings.ADC_ContinuousConvMode = ENABLE;
82 settings.ADC_DataAlign = ADC_DataAlign_Right;
83 settings.ADC_NbrOfConversion = channels.size();
84 ADC_Init(periphInfo.splPtr(), &settings);
85 }
86
87 for (size_t i = 0; i < channels.size(); ++i) {
88 ADC_RegularChannelConfig(periphInfo.splPtr(),
89 channels[i],
90 i + 1,
91 sampleTime);
92 }
93
94 ADC_DMARequestAfterLastTransferCmd(periphInfo.splPtr(), ENABLE);
95
96 dma.config(peripheral::dma::Direction::periphToMem,
97 reinterpret_cast<uintptr_t>(&periphInfo.splPtr()->DR),
98 reinterpret_cast<uintptr_t>(buffer.data()),
99 buffer.size(),
100 sizeof(uint16_t),
101 peripheral::dma::Mode::circular);
102
103 ADC_DMACmd(periphInfo.splPtr(), ENABLE);
104 ADC_Cmd(periphInfo.splPtr(), ENABLE);
105 ADC_SoftwareStartConv(periphInfo.splPtr());
106
107 return true;
108}
