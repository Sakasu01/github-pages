164ErrorStatus SPI_DMA_RW(SPI_TypeDef* SPIx, uint8_t* pData, uint8_t dataSize, SPI_DMA_Mode receive) {
165
166 const SPI_SettingTypedef *SPI_setting;
167 switch ((uint32_t)SPIx) {
168 case (uint32_t)SPI1:
169 SPI_setting = &SPI_settings[0];
170 break;
171 case (uint32_t)SPI2:
172 SPI_setting = &SPI_settings[1];
173 break;
174 case (uint32_t)SPI3:
175 SPI_setting = &SPI_settings[2];
176 break;
177 default:
178 return ERROR;
179 }
180
181 // クロック供給
182 RCC_AHB1PeriphClockCmd(SPI_setting->DMA_tx.RCC_DMA | SPI_setting->DMA_rx.RCC_DMA, ENABLE);
183
184 static DMA_InitTypeDef SPI_DMA_initStructure = {
185 .DMA_PeripheralDataSize = DMA_PeripheralDataSize_Byte,
186 .DMA_MemoryDataSize = DMA_MemoryDataSize_Byte,
187 .DMA_PeripheralInc = DMA_PeripheralInc_Disable,
188 .DMA_Mode = DMA_Mode_Normal,
189 .DMA_Priority = DMA_Priority_VeryHigh,
190 .DMA_FIFOMode = DMA_FIFOMode_Disable,
191 .DMA_FIFOThreshold = DMA_FIFOThreshold_HalfFull,
192 .DMA_MemoryBurst = DMA_MemoryBurst_Single,
193 .DMA_PeripheralBurst = DMA_PeripheralBurst_Single,
194 };
195 uint16_t DummyData = 0xffff;
196
197
198 // shared DMA configuration values
199 SPI_DMA_initStructure.DMA_PeripheralBaseAddr = (uint32_t) (&(SPIx->DR));
200 SPI_DMA_initStructure.DMA_BufferSize = dataSize;
201
202 DMA_DeInit(SPI_setting->DMA_rx.stream);
203 DMA_DeInit(SPI_setting->DMA_tx.stream);
204
205 //受信
206 if (receive == SPI_DMA_Receive) {
207 // Rx
208 SPI_DMA_initStructure.DMA_Memory0BaseAddr = (uint32_t) pData;
209 SPI_DMA_initStructure.DMA_DIR = DMA_DIR_PeripheralToMemory;
210 SPI_DMA_initStructure.DMA_MemoryInc = DMA_MemoryInc_Enable;
211 SPI_DMA_initStructure.DMA_Channel = SPI_setting->DMA_rx.channel;
212 DMA_Init(SPI_setting->DMA_rx.stream, &SPI_DMA_initStructure);
213 // Tx
214 SPI_DMA_initStructure.DMA_Memory0BaseAddr = (uint32_t) (&DummyData);
215 SPI_DMA_initStructure.DMA_DIR = DMA_DIR_MemoryToPeripheral;
216 SPI_DMA_initStructure.DMA_MemoryInc = DMA_MemoryInc_Disable;
217 SPI_DMA_initStructure.DMA_Channel = SPI_setting->DMA_tx.channel;
218 DMA_Init(SPI_setting->DMA_tx.stream, &SPI_DMA_initStructure);
219 } else {
220 // Rx
221 SPI_DMA_initStructure.DMA_Memory0BaseAddr = (uint32_t) (&DummyData);
222 SPI_DMA_initStructure.DMA_DIR = DMA_DIR_PeripheralToMemory;
223 SPI_DMA_initStructure.DMA_MemoryInc = DMA_MemoryInc_Disable;
224 SPI_DMA_initStructure.DMA_Channel = SPI_setting->DMA_rx.channel;
225 DMA_Init(SPI_setting->DMA_rx.stream, &SPI_DMA_initStructure);
226 // Tx
227 SPI_DMA_initStructure.DMA_Memory0BaseAddr = (uint32_t) pData;
228 SPI_DMA_initStructure.DMA_DIR = DMA_DIR_MemoryToPeripheral;
229 SPI_DMA_initStructure.DMA_MemoryInc = DMA_MemoryInc_Enable;
230 SPI_DMA_initStructure.DMA_Channel = SPI_setting->DMA_tx.channel;
231 DMA_Init(SPI_setting->DMA_tx.stream, &SPI_DMA_initStructure);
232 }
233
234 // Enable DMA RX Channel
235 DMA_Cmd(SPI_setting->DMA_rx.stream, ENABLE);
236 // Enable DMA TX Channel
237 DMA_Cmd(SPI_setting->DMA_tx.stream, ENABLE);
238
239 // Enable SPI TX/RX request
240 SPI_I2S_DMACmd(SPIx, SPI_I2S_DMAReq_Tx | SPI_I2S_DMAReq_Rx, ENABLE);
241
242 // Wait until DMA Receive Complete
243 while (DMA_GetFlagStatus(SPI_setting->DMA_rx.stream, SPI_setting->DMA_flag) == RESET)
244 continue;
245
246 // Disable DMA RX Channel
247 DMA_Cmd(SPI_setting->DMA_rx.stream, DISABLE);
248 // Disable DMA TX Channel
249 DMA_Cmd(SPI_setting->DMA_tx.stream, DISABLE);
250
251 // Disable SPI1 RX/TX request
252 SPI_I2S_DMACmd(SPIx, SPI_I2S_DMAReq_Tx | SPI_I2S_DMAReq_Rx, DISABLE);
253
254 SPIx->SR |= SPI_I2S_FLAG_TXE;
255 SPIx->SR &= ~SPI_I2S_FLAG_RXNE;
256
257 return SUCCESS;
258}
