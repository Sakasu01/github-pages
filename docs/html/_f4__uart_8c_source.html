83ErrorStatus UART_config(USART_TypeDef* USARTx, const uint32_t baudrate) {
84 // 初期化用テーブル
85 const UART_SettingTypeDef UART_settings[8] = {
86 UART_SETTING_INITIALIZER(_USART1),
87 UART_SETTING_INITIALIZER(_USART2),
88 UART_SETTING_INITIALIZER(_USART3),
89 UART_SETTING_INITIALIZER(_UART4),
90 UART_SETTING_INITIALIZER(_UART5),
91 UART_SETTING_INITIALIZER(_USART6),
92 UART_SETTING_INITIALIZER(_UART7),
93 UART_SETTING_INITIALIZER(_UART8),
94 };
95
96 UART_SettingTypeDef UART_setting;
97 switch ((uint32_t)USARTx) {
98 case (uint32_t)USART1:
99 UART_setting = UART_settings[0];
100 break;
101 case (uint32_t)USART2:
102 UART_setting = UART_settings[1];
103 break;
104 case (uint32_t)USART3:
105 UART_setting = UART_settings[2];
106 break;
107 case (uint32_t)UART4:
108 UART_setting = UART_settings[3];
109 break;
110 case (uint32_t)UART5:
111 UART_setting = UART_settings[4];
112 break;
113 case (uint32_t)USART6:
114 UART_setting = UART_settings[5];
115 break;
116 case (uint32_t)UART7:
117 UART_setting = UART_settings[6];
118 break;
119 case (uint32_t)UART8:
120 UART_setting = UART_settings[7];
121 break;
122 default:
123 return ERROR;
124 }
125
126 UART_setting.RCC_APBnPeriphClockCmd(UART_setting.RCC_APBnPeriph_USARTx, ENABLE);
127
128 F4_GPIO_config(&UART_setting.txPin, GPIO_High_Speed, GPIO_Mode_AF,
129 GPIO_OType_PP, GPIO_PuPd_NOPULL);
130 F4_GPIO_config(&UART_setting.rxPin, GPIO_High_Speed, GPIO_Mode_AF,
131 GPIO_OType_PP, GPIO_PuPd_UP);
132
133 GPIO_PinAFConfig(UART_setting.txPin.GPIOx, UART_setting.txPin.pinSource,
134 UART_setting.GPIO_AF);
135 GPIO_PinAFConfig(UART_setting.rxPin.GPIOx, UART_setting.rxPin.pinSource,
136 UART_setting.GPIO_AF);
137
138 // USART(UART)初期化
139 USART_InitTypeDef USART_initStructure;
140 USART_initStructure.USART_BaudRate = baudrate; // ボーレートを設定
141 USART_initStructure.USART_WordLength = USART_WordLength_8b; // 8bit転送
142 USART_initStructure.USART_StopBits = USART_StopBits_1; // ストップ1bit
143 USART_initStructure.USART_Parity = USART_Parity_No;
144 USART_initStructure.USART_HardwareFlowControl =
145 USART_HardwareFlowControl_None;
146 USART_initStructure.USART_Mode = USART_Mode_Rx | USART_Mode_Tx;
147 USART_Init(USARTx, &USART_initStructure);
148
149 return SUCCESS;
150}

233uint32_t UART_gets(USART_TypeDef* USARTx, char str[], uint32_t n) {
234 char tmp; // 作業用文字バッファ
235 char *start = str; // 開始位置
236
237 while (n) {
238 tmp = '\0'; // バッファにヌル文字入れといて
239 tmp = UART_getch(USARTx); // 取得した文字入れて
240 if (tmp == '\r' || tmp == '\n') // 改行されるまで
241 break;
242 else if (tmp != '\0') { // 配列に代入し続ける
243 *str = tmp;
244 str++;
245 n--;
246 }
247 }
248
249 *str = '\0'; /* 最後にヌル文字 */
250 return (uint32_t)(str - start);
251}
