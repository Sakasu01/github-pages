78 addr.id = address;
79 CANProtocolMasterCommand_watchDog(Gouyoku_CANx, &addr);
80
81 float buff[] = { encPulse, moduleDiameter, gearRatio, static_cast<float>(motorDirection) }; //要素追加可能
82
83 for (uint8_t i = 0; i < ARRAY_SIZE(buff); i++) {
