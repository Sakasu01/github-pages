64 bool run(const coords::XYT& targetVelo){
65 for (int i = 0; i < N; ++i) {
66 //-xsin0-ycos0+rwsin(0-w)
67 float vx = targetVelo.xy.x() * xCoefficient[i];
68 float vy = targetVelo.xy.y() * yCoefficient[i];
69 float vw = motor.centerToWheel[i] * targetVelo.theta*wCoefficient[i];
