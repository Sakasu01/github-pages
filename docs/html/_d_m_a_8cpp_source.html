19bool DMAMethod::config(peripheral::dma::Direction direction,
20 uintptr_t periphAddr,
21 uintptr_t memoryAddr,
22 std::size_t bufferSize,
23 std::size_t dataSize,
24 peripheral::dma::Mode mode,
25 std::optional<peripheral::dma::InterruptType> interrupt) {
26 bufSize = bufferSize;
27
28 periphInfo.enableClock();
29
30 DMA_Cmd(periphInfo.splPtr(), DISABLE);
31 DMA_DeInit(periphInfo.splPtr());
32 DMA_InitTypeDef settings;
33 DMA_StructInit(&settings);
34 settings.DMA_Channel = channel;
35 settings.DMA_PeripheralBaseAddr = static_cast<uint32_t>(periphAddr);
36 settings.DMA_Memory0BaseAddr = static_cast<uint32_t>(memoryAddr);
37 settings.DMA_DIR = util::enumToReg(direction, DMA_DIR_PeripheralToMemory, DMA_DIR_MemoryToPeripheral);
38 settings.DMA_BufferSize = bufferSize;
39 settings.DMA_PeripheralInc = DMA_PeripheralInc_Disable;
40 settings.DMA_MemoryInc = DMA_MemoryInc_Enable;
41 settings.DMA_Mode = util::enumToReg(mode, DMA_Mode_Normal, DMA_Mode_Circular);
42 settings.DMA_Priority = DMA_Priority_VeryHigh;
43 settings.DMA_FIFOMode = DMA_FIFOMode_Disable;
44 settings.DMA_FIFOThreshold = DMA_FIFOThreshold_Full;
45 settings.DMA_MemoryBurst = DMA_MemoryBurst_Single;
46 settings.DMA_PeripheralBurst = DMA_PeripheralBurst_Single;
47
48 uint32_t dataSizeReg = 0;
49 switch (dataSize) {
50 case 1:
51 dataSizeReg = DMA_PeripheralDataSize_Byte;
52 break;
53 case 2:
54 dataSizeReg = DMA_PeripheralDataSize_HalfWord;
55 break;
56 case 4:
57 dataSizeReg = DMA_PeripheralDataSize_Word;
58 break;
59 default:
60 return false;
61 }
62 settings.DMA_PeripheralDataSize = dataSizeReg;
63 settings.DMA_MemoryDataSize = dataSizeReg;
64
65 DMA_Init(periphInfo.splPtr(), &settings);
66 if (interrupt) {
67 DMA_ITConfig(periphInfo.splPtr(), util::enumToReg(*interrupt, DMA_IT_TC, DMA_IT_HT), ENABLE);
68 }
69 DMA_Cmd(periphInfo.splPtr(), ENABLE);
70 return true;
71}
