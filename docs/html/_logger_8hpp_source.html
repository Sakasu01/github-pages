156 template <class... Args>
157 void log_impl(std::string_view fileName,
158 std::string_view prettyFunc,
159 std::string_view funcName,
160 size_t lineNum,
161 std::string_view message = "",
162 Args&&... args) {
163 elapsedTime = timer.getElapsed();
164 placeholders.message = message;
165 placeholders.fileName = fileName;
166 placeholders.prettyFunc = prettyFunc;
167 placeholders.funcName = funcName;
168 placeholders.lineNum = lineNum;
169 if (fontColor != "") {
170 outputString << fontColor;
171 }
172 output(format.fmt, format.args.cbegin(), std::forward<Args>(args)...);
173 if (fontColor != "") {
174 outputString << fontColorReset;
175 }
176 timeoutCounter.restart();
177 while (outputter.getTxBufferFreeSize() < outputString.str().size()) {
178 if (timeoutMsec && timeoutCounter.getElapsedMsec() > timeoutMsec.value()) {
179 break;
180 }
181 }
182 outputter << outputString.str();
183 outputString.clear();
184 if (autoTimerReset) {
185 timer.restart();
186 }
187 }

195 template <class... Args>
196 void output(std::string_view str, decltype(logger::Format::args)::const_iterator itr, Args&&... args) {
197 auto tail = str.find('$');
198 if (tail == std::string_view::npos || itr == std::end(format.args)) {
199 outputString << str << "\n";
200 return;
201 }
202
203 outputString << str.substr(0, tail);
204 if (*itr == logger::Placeholder::message) {
205 outputString.print(placeholders.message, std::forward<Args>(args)...);
206 }
207 else {
208 outputPlaceholder(*itr);
209 }
210 str.remove_prefix(tail + 1);
211 output(str, itr + 1, std::forward<Args>(args)...);
212 }
