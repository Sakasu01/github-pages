68 bool run(const coords::XYT& targetVelo){
69 for (int i = 0; i < N; ++i) {
70 float vx = targetVelo.xy.x()
71 + targetVelo.theta * motor.wheelDirectionVec[i][0];
72 float vy = targetVelo.xy.y()
73 + targetVelo.theta * motor.wheelDirectionVec[i][1];
74 float vnorm = fastsqrtf(vx * vx + vy * vy);
75 //norm check
76 if (vnorm < epsilon) {
77 drivingVelo[i] = 0.f;
78 steeringAngle[i] = 0.f; //no need?
79 continue;
80 }
81
82 float c = std::cos(steeringAngle[i]);
83 float s = std::sin(steeringAngle[i]);
84 //rotation
85 float wx = c * motor.wheelDirectionVec[i][0] - s * motor.wheelDirectionVec[i][1];
86 float wy = s * motor.wheelDirectionVec[i][0] + c * motor.wheelDirectionVec[i][1];
87
88 //2直線のなす角,内積分の外積
89 float dot = vx * wx + vy * wy;
90 float cross = vx * wy - vy * wx;
91 float sgn = math::sign(dot); //be in range -pi ~ pi
92 //clamp する？
93 steeringAngle[i] -= std::atan2(sgn * cross, sgn * dot);
94
95 for (; steeringAngle[i] < -motor.angleLimit; steeringAngle[i] += motor.angleLimit);
96 for (; steeringAngle[i] > motor.angleLimit; steeringAngle[i] -= motor.angleLimit);
97
98 //CW or CCW?
99 drivingVelo[i] = math::sign(dot) * vnorm;
100 }
101 //norm and angle output
102 applyAngle();
103 applyVelo();
104 return true;
105 }
