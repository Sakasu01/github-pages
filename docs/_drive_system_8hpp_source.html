96 void update(){
97 if (currentStatus == Status::OFF) return;
98
99 coords::FieldCoords fc = dr.getFieldCoords();
100
101 // Manual or Auto
102 coords::XYT targetVelo;
103 if (currentStatus == Status::MANUAL) {
104 targetVelo = manual.getTargetVelo(fc);
105 } else if (currentStatus == Status::RUNNING) {
106 // posを取って
107 const coords::FieldCoords &ref = route.getTargetFieldCoords(time_ms);
108 // controlに渡して出力
109 targetVelo = control.calcNextVelo(fc, ref);
110
111 coords::XYT error = fc.pos - ref.pos;
112 if (error.xy.norm() > divergenceRadius
113 || std::abs(error.theta) > divergenceAngle) {
114 currentStatus = Status::ERROR;
115 }
116// mother::serial << fc.pos.xy.x() << " " << fc.pos.xy.y() << " ";
117// mother::serial << ref.pos.xy.x() << " " << ref.pos.xy.y() << "\n";
118 //if分岐で収束判定
119 if (!route.isPlaying) {
120 if (error.xy.norm() < convergenceRadius &&
121 std::abs(error.theta) < convergenceAngle)
122 {
123 currentStatus = Status::CONVERGENCE;
124 targetVelo = { { 0.f, 0.f }, 0.f };
125 }
126 }
127 } else if (currentStatus == Status::CONVERGENCE) {
128 control.reset();
129 }
130
131 //to kinematics
132 if (currentStatus == Status::READY) {
133 if (route.getFirstFieldVeloNorm() > 1e-5f) { //TODO cash
134 kinematics.faceTo(route.getFirstFieldDirection());
135 }
136 } else if (currentStatus == Status::ERROR) {
137 kinematics.errorHandle();
138 } else {
139 if (!kinematics.run(targetVelo)) {
140 currentStatus = Status::ERROR;
141 }
142 }
143 time_ms += updateCycleTime;
144 }
