40/* Definitions for MMC/SDC command */
41#define CMD0 (0x40+0) /* GO_IDLE_STATE */
42#define CMD1 (0x40+1) /* SEND_OP_COND (MMC) */
43#define ACMD41 (0xC0+41) /* SEND_OP_COND (SDC) */
44#define CMD8 (0x40+8) /* SEND_IF_COND */
45#define CMD9 (0x40+9) /* SEND_CSD */
46#define CMD10 (0x40+10) /* SEND_CID */
47#define CMD12 (0x40+12) /* STOP_TRANSMISSION */
48#define ACMD13 (0xC0+13) /* SD_STATUS (SDC) */
49#define CMD16 (0x40+16) /* SET_BLOCKLEN */
50#define CMD17 (0x40+17) /* READ_SINGLE_BLOCK */
51#define CMD18 (0x40+18) /* READ_MULTIPLE_BLOCK */
52#define CMD23 (0x40+23) /* SET_BLOCK_COUNT (MMC) */
53#define ACMD23 (0xC0+23) /* SET_WR_BLK_ERASE_COUNT (SDC) */
54#define CMD24 (0x40+24) /* WRITE_BLOCK */
55#define CMD25 (0x40+25) /* WRITE_MULTIPLE_BLOCK */

92enum speed_setting { INTERFACE_SLOW, INTERFACE_FAST };
93
94static void interface_speed( enum speed_setting speed )
95{
96 DWORD tmp;
97
98 tmp = SPI_SD->CR1;
99 if ( speed == INTERFACE_SLOW ) {
100 /* Set slow clock (100k-400k) */
101 tmp = ( tmp | SPI_BaudRatePrescaler_256 );
102 } else {
103 /* Set fast clock (depends on the CSD) */
104 tmp = ( tmp & ~SPI_BaudRatePrescaler_256 ) | SPI_BaudRatePrescaler_SPI_SD;
105 }
106 SPI_SD->CR1 = tmp;
107}

218/*-----------------------------------------------------------------------*/
219/* Transmit/Receive a byte to MMC via SPI (Platform dependent) */
220/*-----------------------------------------------------------------------*/
221static BYTE stm32_spi_rw( BYTE out )
222{
223 BYTE tmp;
224 /* Loop while DR register in not empty */
226
227 /* Send byte through the SPI1 peripheral */
228 SPI_I2S_SendData(SPI_SD, out);
229
230 /* Wait to receive a byte */
231 while (SPI_I2S_GetFlagStatus(SPI_SD, SPI_I2S_FLAG_RXNE) == RESET) { ; }
232
233 tmp = SPI_I2S_ReceiveData(SPI_SD);
234//UART_printf(DEBUG_USARTx,"%d\n",tmp);
235 /* Return the byte read from the SPI bus */
236 return tmp;
237}

267BYTE wait_ready (void)
268{
269 BYTE res;
270
271
272 Timer2 = 50; /* Wait for ready in timeout of 500ms */
273 rcvr_spi();

275 res = rcvr_spi();

294#ifdef STM32_USE_DMA
295/*-----------------------------------------------------------------------*/
296/* Transmit/Receive Block using DMA (Platform dependent. STM32 here) */
297/*-----------------------------------------------------------------------*/
298static
299void stm32_dma_transfer(
300 bool receive, /* false for buff->SPI, true for SPI->buff */
301 const BYTE *buff, /* receive true : 512 byte data block to be transmitted
302 receive false : Data buffer to store received data */
303 UINT btr /* receive true : Byte count (must be multiple of 2)
304 receive false : Byte count (must be 512) */
305)
306{
307
308 DMA_InitTypeDef DMA_InitStructure;
309 WORD rw_workbyte[] = { 0xffff };
310
311 /* shared DMA configuration values */
312 DMA_InitStructure.DMA_PeripheralBaseAddr = (DWORD)(&(SPI_SD->DR));
313 DMA_InitStructure.DMA_PeripheralDataSize = DMA_PeripheralDataSize_Byte;
314 DMA_InitStructure.DMA_MemoryDataSize = DMA_MemoryDataSize_Byte;
315 DMA_InitStructure.DMA_PeripheralInc = DMA_PeripheralInc_Disable;
316 DMA_InitStructure.DMA_BufferSize = btr;
317 DMA_InitStructure.DMA_Mode = DMA_Mode_Normal;
318 DMA_InitStructure.DMA_Priority = DMA_Priority_VeryHigh;
319
320 DMA_DeInit(DMA_Stream_SPI_SD_RX);
321 DMA_DeInit(DMA_Stream_SPI_SD_TX);
322
323 if ( receive ) {
324
325 /* DMA1 channel2 configuration SPI1 RX ---------------------------------------------*/
326 /* DMA1 channel4 configuration SPI2 RX ---------------------------------------------*/
327 DMA_InitStructure.DMA_Memory0BaseAddr = (DWORD)buff;
328 DMA_InitStructure.DMA_DIR = DMA_DIR_PeripheralToMemory;
329 DMA_InitStructure.DMA_MemoryInc = DMA_MemoryInc_Enable;
330 DMA_InitStructure.DMA_Channel = DMA_Channel_SPI_SD_RX;
331 DMA_Init(DMA_Stream_SPI_SD_RX, &DMA_InitStructure);
332
333 /* DMA1 channel3 configuration SPI1 TX ---------------------------------------------*/
334 /* DMA1 channel5 configuration SPI2 TX ---------------------------------------------*/
335 DMA_InitStructure.DMA_Memory0BaseAddr = (DWORD)rw_workbyte;
336 DMA_InitStructure.DMA_DIR = DMA_DIR_MemoryToPeripheral;
337 DMA_InitStructure.DMA_MemoryInc = DMA_MemoryInc_Disable;
338 DMA_InitStructure.DMA_Channel = DMA_Channel_SPI_SD_TX;
339 DMA_Init(DMA_Stream_SPI_SD_TX, &DMA_InitStructure);
340
341 } else {
342
343 /* DMA1 channel2 configuration SPI1 RX ---------------------------------------------*/
344 /* DMA1 channel4 configuration SPI2 RX ---------------------------------------------*/
345 DMA_InitStructure.DMA_Memory0BaseAddr = (DWORD)rw_workbyte;
346 DMA_InitStructure.DMA_DIR = DMA_DIR_PeripheralToMemory;
347 DMA_InitStructure.DMA_MemoryInc = DMA_MemoryInc_Disable;
348 DMA_InitStructure.DMA_Channel = DMA_Channel_SPI_SD_RX;
349 DMA_Init(DMA_Stream_SPI_SD_RX, &DMA_InitStructure);
350
351 /* DMA1 channel3 configuration SPI1 TX ---------------------------------------------*/
352 /* DMA1 channel5 configuration SPI2 TX ---------------------------------------------*/
353 DMA_InitStructure.DMA_Memory0BaseAddr = (DWORD)buff;
354 DMA_InitStructure.DMA_DIR = DMA_DIR_MemoryToPeripheral;
355 DMA_InitStructure.DMA_MemoryInc = DMA_MemoryInc_Enable;
356 DMA_InitStructure.DMA_Channel = DMA_Channel_SPI_SD_TX;
357 DMA_Init(DMA_Stream_SPI_SD_TX, &DMA_InitStructure);
358
359 }
360
361 /* Enable DMA RX Channel */
362 DMA_Cmd(DMA_Stream_SPI_SD_RX, ENABLE);
363 /* Enable DMA TX Channel */
364 DMA_Cmd(DMA_Stream_SPI_SD_TX, ENABLE);
365
366 /* Enable SPI TX/RX request */
367 SPI_I2S_DMACmd(SPI_SD, SPI_I2S_DMAReq_Rx | SPI_I2S_DMAReq_Tx, ENABLE);
368
369 /* Wait until DMA1_Channel 2 Receive Complete */
370 while (DMA_GetFlagStatus(DMA_Stream_SPI_SD_RX, DMA_FLAG_SPI_SD_TC_RX) == RESET) { ; }
371
372 /* Disable DMA RX Channel */
373 DMA_Cmd(DMA_Stream_SPI_SD_RX, DISABLE);
374 /* Disable DMA TX Channel */
375 DMA_Cmd(DMA_Stream_SPI_SD_TX, DISABLE);
376
377 /* Disable SPI1 RX/TX request */
378 SPI_I2S_DMACmd(SPI_SD, SPI_I2S_DMAReq_Rx | SPI_I2S_DMAReq_Tx, DISABLE);
379}
380#endif /* STM32_USE_DMA */

383/*-----------------------------------------------------------------------*/
384/* Power Control (Platform dependent) */
385/*-----------------------------------------------------------------------*/
386/* When the target system does not support socket power control, there */
387/* is nothing to do in these functions and chk_power always returns 1. */
388
389static
390void power_on (void)
391{
392 SPI_InitTypeDef SPI_InitStructure;
393 GPIO_InitTypeDef GPIO_InitStructure;
394 volatile BYTE dummyread;
395
396 /* Enable GPIO clock for CS */
397 RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIO_CS, ENABLE);
398 /* Enable SPI clock, SPI1-APB2, SPI2-APB1 */
399 RCC_APBPeriphClockCmd_SPI_SD(RCC_APBPeriph_SPI_SD, ENABLE);
400
401 card_power(1);
402 socket_wp_cp_init();
403
404 for (Timer1 = 25; Timer1; ); /* Wait for 250ms */
405
406 /* Configure I/O for Flash Chip select */
407 GPIO_InitStructure.GPIO_Pin = GPIO_Pin_CS;
408 GPIO_InitStructure.GPIO_Mode = GPIO_Mode_OUT;
409 GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
410 GPIO_InitStructure.GPIO_Speed = GPIO_Speed_100MHz;
411 GPIO_Init(GPIO_CS, &GPIO_InitStructure);
412 /* Deselect the Card: Chip Select high */
413 DESELECT();
414
415 /* Configure SPI pins: SCK and MOSI with default alternate function (not remapped) push-pull */
416 GPIO_InitStructure.GPIO_Pin = GPIO_Pin_SPI_SD_SCK | GPIO_Pin_SPI_SD_MISO | GPIO_Pin_SPI_SD_MOSI;
417 GPIO_InitStructure.GPIO_Speed = GPIO_Speed_100MHz;
418 GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;
419// GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
420 GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
421 GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;
422 GPIO_Init(GPIO_SPI_SD, &GPIO_InitStructure);
423
424// /* Configure MISO as Input with internal pull-up */
425// GPIO_InitStructure.GPIO_Pin = GPIO_Pin_SPI_SD_MISO;
426// GPIO_InitStructure.GPIO_Mode = GPIO_Mode_IN;
427// GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;
428// GPIO_Init(GPIO_SPI_SD, &GPIO_InitStructure);
429
430 GPIO_PinAFConfig(GPIO_CS, GPIO_PinSource_Pin_CS,SPI_AF_SD);
431 GPIO_PinAFConfig(GPIO_SPI_SD,GPIO_PinSource_SPI_SD_SCK,SPI_AF_SD);
432 GPIO_PinAFConfig(GPIO_SPI_SD,GPIO_PinSource_SPI_SD_MOSI,SPI_AF_SD);
433 GPIO_PinAFConfig(GPIO_SPI_SD,GPIO_PinSource_SPI_SD_MISO,SPI_AF_SD);
434
435 /* SPI1 configuration */
436 SPI_InitStructure.SPI_Direction = SPI_Direction_2Lines_FullDuplex;
437 SPI_InitStructure.SPI_Mode = SPI_Mode_Master;
438 SPI_InitStructure.SPI_DataSize = SPI_DataSize_8b;
439 SPI_InitStructure.SPI_CPOL = SPI_CPOL_Low;
440 SPI_InitStructure.SPI_CPHA = SPI_CPHA_1Edge;
441 SPI_InitStructure.SPI_NSS = SPI_NSS_Soft;
442 SPI_InitStructure.SPI_BaudRatePrescaler = SPI_BaudRatePrescaler_256; // 72000kHz/256=281kHz < 400Hz
443 SPI_InitStructure.SPI_FirstBit = SPI_FirstBit_MSB;
444 SPI_InitStructure.SPI_CRCPolynomial = 7;
445 SPI_Init(SPI_SD, &SPI_InitStructure);
446
447 SPI_CalculateCRC(SPI_SD, DISABLE);
448
449 /* Enable SPIx */
450 SPI_Cmd(SPI_SD, ENABLE);
451
452 /* drain SPI */
453 while (SPI_I2S_GetFlagStatus(SPI_SD, SPI_I2S_FLAG_TXE) == RESET) { ; }
454 dummyread = SPI_I2S_ReceiveData(SPI_SD);
455
456// UART_printf(DEBUG_USARTx,"%d",dummyread);
457#ifdef STM32_USE_DMA
458 /* enable DMA clock */
459 RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_DMA1, ENABLE);
460#endif
461}

463static
464void power_off (void)
465{
466 GPIO_InitTypeDef GPIO_InitStructure;
467
468 if (!(Stat & STA_NOINIT)) {
469 SELECT();
470 wait_ready();
471 release_spi();
472 }
473
474 SPI_Cmd(SPI_SD, DISABLE);
475 SPI_I2S_DeInit(SPI_SD);
476
477 RCC_APBPeriphClockCmd_SPI_SD(RCC_APBPeriph_SPI_SD, DISABLE);
478
479 /* All SPI-Pins to input with weak internal pull-downs */
480 GPIO_InitStructure.GPIO_Pin = GPIO_Pin_SPI_SD_SCK | GPIO_Pin_SPI_SD_MISO | GPIO_Pin_SPI_SD_MOSI;
481 GPIO_InitStructure.GPIO_Mode = GPIO_Mode_IN;
482 GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_DOWN;
483 GPIO_Init(GPIO_SPI_SD, &GPIO_InitStructure);
484
485 card_power(0);
486
487 Stat |= STA_NOINIT; /* Set STA_NOINIT */
488}

491/*-----------------------------------------------------------------------*/
492/* Receive a data packet from MMC */
493/*-----------------------------------------------------------------------*/
494
495static
496bool rcvr_datablock (
497 BYTE *buff, /* Data buffer to store received data */
498 UINT btr /* Byte count (must be multiple of 4) */
499)
500{
501 BYTE token;
502
503
504 Timer1 = 10;
505 do { /* Wait for data packet in timeout of 100ms */
506 token = rcvr_spi();
507 } while ((token == 0xFF) && Timer1);
508 if(token != 0xFE) return false; /* If not valid data token, return with error */
509
510#ifdef STM32_USE_DMA
511 stm32_dma_transfer( true, buff, btr );
512#else
513 do { /* Receive the data block into buffer */
514 rcvr_spi_m(buff++);
515 rcvr_spi_m(buff++);
516 rcvr_spi_m(buff++);
517 rcvr_spi_m(buff++);
518 } while (btr -= 4);
519#endif /* STM32_USE_DMA */
520
521 rcvr_spi(); /* Discard CRC */
522 rcvr_spi();
523
524 return true; /* Return with success */
525}

533#if _READONLY == 0
534static
535bool xmit_datablock (
536 const BYTE *buff, /* 512 byte data block to be transmitted */
537 BYTE token /* Data/Stop token */
538)
539{
540 BYTE resp;
541#ifndef STM32_USE_DMA
542 BYTE wc;
543#endif
544
545 if (wait_ready() != 0xFF) return false;
546
547 xmit_spi(token); /* Xmit data token */
548 if (token != 0xFD) { /* Is data token */
549
550#ifdef STM32_USE_DMA
551 stm32_dma_transfer( false, buff, 512 );
552#else
553 wc = 0;
554 do { /* Xmit the 512 byte data block to MMC */
555 xmit_spi(*buff++);
556 xmit_spi(*buff++);
557 } while (--wc);
558#endif /* STM32_USE_DMA */
559
560 xmit_spi(0xFF); /* CRC (Dummy) */
561 xmit_spi(0xFF);
562 resp = rcvr_spi(); /* Receive data response */
563 if ((resp & 0x1F) != 0x05) /* If not accepted, return with error */
564 return false;
565 }
566
567 return true;
568}

573/*-----------------------------------------------------------------------*/
574/* Send a command packet to MMC */
575/*-----------------------------------------------------------------------*/
576
577static
578BYTE send_cmd (
579 BYTE cmd, /* Command byte */
580 DWORD arg /* Argument */
581)
582{
583 BYTE n, res;
584
585
586 if (cmd & 0x80) { /* ACMD<n> is the command sequence of CMD55-CMD<n> */
587 cmd &= 0x7F;
588 res = send_cmd(CMD55, 0);
589 if (res > 1) return res;
590 }
591
592 /* Select the card and wait for ready */
593 DESELECT();
594 SELECT();
595 if (wait_ready() != 0xFF) {
596 return 0xFF;
597 }
598
599 /* Send command packet */
600 xmit_spi(cmd); /* Start + Command index */
601 xmit_spi((BYTE)(arg >> 24)); /* Argument[31..24] */
602 xmit_spi((BYTE)(arg >> 16)); /* Argument[23..16] */
603 xmit_spi((BYTE)(arg >> 8)); /* Argument[15..8] */
604 xmit_spi((BYTE)arg); /* Argument[7..0] */
605 n = 0x01; /* Dummy CRC + Stop */
606 if (cmd == CMD0) n = 0x95; /* Valid CRC for CMD0(0) */
607 if (cmd == CMD8) n = 0x87; /* Valid CRC for CMD8(0x1AA) */
608 xmit_spi(n);
609
610 /* Receive command response */
611 if (cmd == CMD12) rcvr_spi(); /* Skip a stuff byte when stop reading */
612
613 n = 10; /* Wait for a valid response in timeout of 10 attempts */
614 do
615 res = rcvr_spi();
616 while ((res & 0x80) && --n);
617
618 return res; /* Return with the response value */
619}

630/*-----------------------------------------------------------------------*/
631/* Initialize Disk Drive */
632/*-----------------------------------------------------------------------*/
633
634DSTATUS SD_SPI_disk_initialize ()
635{
636 BYTE n, cmd, ty, ocr[4];
637
638 if (Stat & STA_NODISK) return Stat; /* No card in the socket */
639
640 power_on(); /* Force socket power on */
641 interface_speed(INTERFACE_SLOW);
642 for (n = 10; n; n--) rcvr_spi(); /* 80 dummy clocks */
643
644 ty = 0;
645 if (send_cmd(CMD0, 0) == 1) { /* Enter Idle state */
646 Timer1 = 100; /* Initialization timeout of 1000 msec */
647 if (send_cmd(CMD8, 0x1AA) == 1) { /* SDHC */
648 for (n = 0; n < 4; n++) ocr[n] = rcvr_spi(); /* Get trailing return value of R7 resp */
649 if (ocr[2] == 0x01 && ocr[3] == 0xAA) { /* The card can work at vdd range of 2.7-3.6V */
650 while (Timer1 && send_cmd(ACMD41, 1UL << 30)); /* Wait for leaving idle state (ACMD41 with HCS bit) */
651 if (Timer1 && send_cmd(CMD58, 0) == 0) { /* Check CCS bit in the OCR */
652 for (n = 0; n < 4; n++) ocr[n] = rcvr_spi();
653 ty = (ocr[0] & 0x40) ? CT_SD2 | CT_BLOCK : CT_SD2;
654 }
655 }
656 } else { /* SDSC or MMC */
657 if (send_cmd(ACMD41, 0) <= 1) {
658 ty = CT_SD1; cmd = ACMD41; /* SDSC */
659 } else {
660 ty = CT_MMC; cmd = CMD1; /* MMC */
661 }
662 while (Timer1 && send_cmd(cmd, 0)); /* Wait for leaving idle state */
663 if (!Timer1 || send_cmd(CMD16, 512) != 0) /* Set R/W block length to 512 */
664 ty = 0;
665 }
666 }
667 CardType = ty;
668 release_spi();
669
670 if (ty) { /* Initialization succeeded */
671 Stat &= ~STA_NOINIT; /* Clear STA_NOINIT */
672 interface_speed(INTERFACE_FAST);
673 } else { /* Initialization failed */
674 power_off();
675 }
676
677 return Stat;
678}

693/*-----------------------------------------------------------------------*/
694/* Read Sector(s) */
695/*-----------------------------------------------------------------------*/
696
697DRESULT SD_SPI_disk_read (
698 BYTE *buff, /* Pointer to the data buffer to store read data */
699 DWORD sector, /* Start sector number (LBA) */
700 BYTE count /* Sector count (1..255) */
701)
702{
703 if (!count) return RES_PARERR;
704 if (Stat & STA_NOINIT) return RES_NOTRDY;
705
706 if (!(CardType & CT_BLOCK)) sector *= 512; /* Convert to byte address if needed */
707
708 if (count == 1) { /* Single block read */
709 if (send_cmd(CMD17, sector) == 0) { /* READ_SINGLE_BLOCK */
710 if (rcvr_datablock(buff, 512)) {
711 count = 0;
712 }
713 }
714 }
715 else { /* Multiple block read */
716 if (send_cmd(CMD18, sector) == 0) { /* READ_MULTIPLE_BLOCK */
717 do {
718 if (!rcvr_datablock(buff, 512)) {
719 break;
720 }
721 buff += 512;
722 } while (--count);
723 send_cmd(CMD12, 0); /* STOP_TRANSMISSION */
724 }
725 }
726 release_spi();
727
728 return count ? RES_ERROR : RES_OK;
729}

738DRESULT SD_SPI_disk_write (
739 const BYTE *buff, /* Data to be written */
740 DWORD sector, /* Start sector number (LBA) */
741 UINT count /* Number of sectors to write (1..128) */
742)
743{

747
748 if (!(CardType & CT_BLOCK)) sector *= 512; /* Convert to byte address if needed */
749
750 if (count == 1) { /* Single block write */
751 if ((send_cmd(CMD24, sector) == 0) /* WRITE_BLOCK */
752 && xmit_datablock(buff, 0xFE))
753 count = 0;
754 }
755 else { /* Multiple block write */
756 if (CardType & CT_SDC) send_cmd(ACMD23, count);
757 if (send_cmd(CMD25, sector) == 0) { /* WRITE_MULTIPLE_BLOCK */
758 do {
759 if (!xmit_datablock(buff, 0xFC)) break;
760 buff += 512;
761 } while (--count);
762 if (!xmit_datablock(0, 0xFD)) /* STOP_TRAN token */
763 count = 1;
764 }
765 }
766 release_spi();
767
768 return count ? RES_ERROR : RES_OK;

777
778#if _USE_IOCTL != 0
779DRESULT SD_SPI_disk_ioctl (
780 BYTE ctrl, /* Control code */
781 void *buff /* Buffer to send/receive control data */
782)
783{
784 DRESULT res;
785 BYTE n, csd[16], *ptr = buff;
786 WORD csize;
787
788 res = RES_ERROR;
789
790 if (ctrl == CTRL_POWER) {
791 switch (*ptr) {
792 case 0: /* Sub control code == 0 (POWER_OFF) */
793 if (chk_power())
794 power_off(); /* Power off */
795 res = RES_OK;
796 break;
797 case 1: /* Sub control code == 1 (POWER_ON) */
798 power_on(); /* Power on */
799 res = RES_OK;
800 break;
801 case 2: /* Sub control code == 2 (POWER_GET) */
802 *(ptr+1) = (BYTE)chk_power();
803 res = RES_OK;
804 break;
805 default :
806 res = RES_PARERR;
807 break;
808 }
809 }
810 else {
811 if (Stat & STA_NOINIT) return RES_NOTRDY;
812
813 switch (ctrl) {
814 case CTRL_SYNC : /* Make sure that no pending write process */
815 SELECT();
816 if (wait_ready() == 0xFF)
817 res = RES_OK;
818 break;
819
820 case GET_SECTOR_COUNT : /* Get number of sectors on the disk (DWORD) */
821 if ((send_cmd(CMD9, 0) == 0) && rcvr_datablock(csd, 16)) {
822 if ((csd[0] >> 6) == 1) { /* SDC ver 2.00 */
823 csize = csd[9] + ((WORD)csd[8] << 8) + 1;
824 *(DWORD*)buff = (DWORD)csize << 10;
825 } else { /* SDC ver 1.XX or MMC*/
826 n = (csd[5] & 15) + ((csd[10] & 128) >> 7) + ((csd[9] & 3) << 1) + 2;
827 csize = (csd[8] >> 6) + ((WORD)csd[7] << 2) + ((WORD)(csd[6] & 3) << 10) + 1;
828 *(DWORD*)buff = (DWORD)csize << (n - 9);
829 }
830 res = RES_OK;
831 }
832 break;
833
834 case GET_SECTOR_SIZE : /* Get R/W sector size (WORD) */
835 *(WORD*)buff = 512;
836 res = RES_OK;
837 break;
838
839 case GET_BLOCK_SIZE : /* Get erase block size in unit of sector (DWORD) */
840 if (CardType & CT_SD2) { /* SDC ver 2.00 */
841 if (send_cmd(ACMD13, 0) == 0) { /* Read SD status */
842 rcvr_spi();
843 if (rcvr_datablock(csd, 16)) { /* Read partial block */
844 for (n = 64 - 16; n; n--) rcvr_spi(); /* Purge trailing data */
845 *(DWORD*)buff = 16UL << (csd[10] >> 4);
846 res = RES_OK;
847 }
848 }
849 } else { /* SDC ver 1.XX or MMC */
850 if ((send_cmd(CMD9, 0) == 0) && rcvr_datablock(csd, 16)) { /* Read CSD */
851 if (CardType & CT_SD1) { /* SDC ver 1.XX */
852 *(DWORD*)buff = (((csd[10] & 63) << 1) + ((WORD)(csd[11] & 128) >> 7) + 1) << ((csd[13] >> 6) - 1);
853 } else { /* MMC */
854 *(DWORD*)buff = ((WORD)((csd[10] & 124) >> 2) + 1) * (((csd[11] & 3) << 3) + ((csd[11] & 224) >> 5) + 1);
855 }
856 res = RES_OK;
857 }
858 }
859 break;
860
861 case MMC_GET_TYPE : /* Get card type flags (1 byte) */
862 *ptr = CardType;
863 res = RES_OK;
864 break;
865
866 case MMC_GET_CSD : /* Receive CSD as a data block (16 bytes) */
867 if (send_cmd(CMD9, 0) == 0 /* READ_CSD */
868 && rcvr_datablock(ptr, 16))
869 res = RES_OK;
870 break;
871
872 case MMC_GET_CID : /* Receive CID as a data block (16 bytes) */
873 if (send_cmd(CMD10, 0) == 0 /* READ_CID */
874 && rcvr_datablock(ptr, 16))
875 res = RES_OK;
876 break;
877
878 case MMC_GET_OCR : /* Receive OCR as an R3 resp (4 bytes) */
879 if (send_cmd(CMD58, 0) == 0) { /* READ_OCR */
880 for (n = 4; n; n--) *ptr++ = rcvr_spi();
881 res = RES_OK;
882 }
883 break;
884
885 case MMC_GET_SDSTAT : /* Receive SD status as a data block (64 bytes) */
886 if (send_cmd(ACMD13, 0) == 0) { /* SD_STATUS */
887 rcvr_spi();
888 if (rcvr_datablock(ptr, 64))
889 res = RES_OK;
890 }
891 break;
892
893 default:
894 res = RES_PARERR;
895 break;
896 }
897
898 release_spi();
899 }
900
901 return res;
902}
903#endif /* _USE_IOCTL != 0 */

909/* This function must be called in period of 10ms */
910
911void disk_timerproc (void)
912{
913 static socket_state_t pv;
914 socket_state_t ns;
915 BYTE n, s;
916
917
918 n = Timer1; /* 100Hz decrement timer */
919 if (n) Timer1 = --n;
920 n = Timer2;
921 if (n) Timer2 = --n;
922
923 ns = pv;
924 pv = socket_wp_cp_state(); /* Sample socket switch */
925
926 if (ns == pv) { /* Have contacts stabled? */
927 s = Stat;
928
929 if (socket_is_write_protected(pv)) /* WP is H (write protected) */
930 s |= STA_PROTECT;
931 else /* WP is L (write enabled) */
932 s &= ~STA_PROTECT;
933
934 if (socket_is_empty(pv)) /* INS = H (Socket empty) */
935 s |= (STA_NODISK | STA_NOINIT);
936 else /* INS = L (Card inserted) */
937 s &= ~STA_NODISK;
938
939 Stat = s;
940 }
