43bool CANMethod::config(peripheral::Pin txPin, peripheral::Pin rxPin,
44 peripheral::can::Mode mode, uint32_t bps) {
45 RCC_APB2PeriphClockCmd(RCC_APB2Periph_SYSCFG, ENABLE);
46 periphInfo.enableClock();
47
48 auto txAf = getAF(periphInfo.baseAddr);
49 auto rxAf = getAF(periphInfo.baseAddr);
50
51 GPIO tx(txPin), rx(rxPin);
52 tx.config(GPIO::Mode::alternate);
53 tx.configAF(txAf);
54 rx.config(GPIO::Mode::alternate);
55 rx.configAF(rxAf);
56
57 CAN_InitTypeDef settings;
58 CAN_StructInit(&settings);
59 settings.CAN_Mode = util::enumToReg(mode, CAN_Mode_Normal, CAN_Mode_Silent);
60 // 再同期を行う際のビット幅を時間単位で指定
61 settings.CAN_SJW = CAN_SJW_1tq;
62
63#if defined(STM32F427_437xx) || defined(STM32F429_439xx) || defined(STM32F446xx) || defined(STM32F469_479xx)
64 // 2018/10/26 F446REなどベース周波数が180MHzのマイコンはBS1,2の組み合わせが2,3だと1MHzちょうどが出せないので、
65 // 無理やりですがifdefで分けることにします
66 // BS2の設定が変わってます
67 // 今手元で確認した感じだと通信できている...と思う
68
69 // レジスタ的には-1するべきですが、CAN_initの中で-1されているので、そのまま計算します
70 // 5 := 1+BS1+BS2
71 settings.CAN_Prescaler = periphInfo.baseFrequency / bps / 5;
72 // ビットタイミングレジスタ
73 settings.CAN_BS1 = CAN_BS1_2tq;
74 settings.CAN_BS2 = CAN_BS2_2tq;
75#else
76 // レジスタ的には-1するべきですが、CAN_initの中で-1されているので、そのまま計算します
77 // 6 := 1+BS1+BS2
78 settings.CAN_Prescaler = periphInfo.baseFrequency / bps / 6;
79 // ビットタイミングレジスタ
80 settings.CAN_BS1 = CAN_BS1_2tq;
81 settings.CAN_BS2 = CAN_BS2_3tq;
82#endif
83
84 // タイムトリガモード(ハードウェアでタイムスタンプを生成する)
85 settings.CAN_TTCM = DISABLE;
86 // エラー時の自動バスオフ設定(ENABLE:自動でバスオフから戻る)
87 settings.CAN_ABOM = ENABLE;
88 // 自動ウェイクアップ復帰モード(DISABLE:ソフトウェアによって復帰)
89 settings.CAN_AWUM = DISABLE;
90 // 自動再送信禁止設定(DISABLE:再送信が有効)
91 settings.CAN_NART = DISABLE;
92 // FIFOのオーバーラン時にメッセージを破棄するか(DISABLEで上書き)
93 settings.CAN_RFLM = DISABLE;
94 // 送信バッファにたまったデータの送信順序(ENABLE:FIFO DISABLE:SID順)
95 settings.CAN_TXFP = ENABLE;
96
97 CAN_DeInit(periphInfo.splPtr());
98 if (CAN_Init(periphInfo.splPtr(), &settings) == CAN_InitStatus_Failed) {
99 return false;
100 }
101 if (CAN_WakeUp(periphInfo.splPtr()) == CAN_WakeUp_Failed) {
102 return false;
103 }
104
105 callBackObj[periphInfo.splPtr() == CAN1 ? 0 : 1] = this;
106
107 return true;
108}

117bool CANMethod::appendFilter(uint8_t num, peripheral::can::Message::IDType idType,
118 uint32_t mask, uint32_t mach) {
119
120 if (13 < num) {
121 return false;
122 }
123
124 // 2018.8.10追記
125 // どうやらF4のCANフィルタは28個あるのだが、そのうち0-13をCAN1,14-27をCAN2が使うようになっているらしい
126 // (この境目はCAN1->FMRをいじることでソフトウェアから変更可能)
127 // 問題なのは、その事実がリファレンスからは非常に読み取りづらいこと(FMRレジスタの項にちょっとだけ記述がある)
128 // (ymdが重要なセクションを読み飛ばしている可能性は否定できない)
129 if (periphInfo.baseAddr == CAN2_BASE) {
130 num += 14;
131 }
132
133 // 細かい事はf3のリファレンスのp.1027あたりを参照のこと
134 CAN_FilterInitTypeDef settings;
135 // どのフィルタについて設定するか ちなみに0～13まで設定可能
136 settings.CAN_FilterNumber = num;
137 // フィルタモード
138 settings.CAN_FilterMode = CAN_FilterMode_IdMask;
139 // フィルタサイズ
140 settings.CAN_FilterScale = CAN_FilterScale_32bit;
141 // フィルタ通過後のデータをFIFOのどれに割り振るか
142 settings.CAN_FilterFIFOAssignment = CAN_FilterFIFO0;
143 if (idType == peripheral::can::Message::IDType::Standard) {
144 mask &= 0x7FF;
145 mach &= 0x7FF;
146
147 // マスク
148 settings.CAN_FilterMaskIdHigh = mask << 5;
149 settings.CAN_FilterMaskIdLow = 0;
150 // 比較値
151 settings.CAN_FilterIdHigh = mask << 5;
152 settings.CAN_FilterIdLow = 0;
153 } else {
154 mask &= 0x1FFFFFFF;
155 mach &= 0x1FFFFFFF;
156
157 // マスク
158 settings.CAN_FilterMaskIdHigh = mask >> 13;
159 settings.CAN_FilterMaskIdLow = mask << 3;
160 // 比較値
161 settings.CAN_FilterIdHigh = mach >> 13;
162 settings.CAN_FilterIdLow = mach << 3;
163 }
164 // フィルタの有効、無効
165 settings.CAN_FilterActivation = ENABLE;
166 CAN_FilterInit(&settings);
167
168 return true;
169}

171bool CANMethod::sendMessage(const peripheral::can::Message &msg) {
172 CanTxMsg txMsg;
173 txMsg.StdId = msg.id;
174 txMsg.ExtId = msg.id;
175 txMsg.IDE = util::enumToReg(msg.idType, CAN_Id_Standard, CAN_Id_Extended);
176 txMsg.RTR = util::enumToReg(msg.frameType, CAN_RTR_Data, CAN_RTR_Remote);
177 txMsg.DLC = msg.data.size();
178 for (uint32_t i = 0; i < txMsg.DLC; ++i) {
179 txMsg.Data[i] = static_cast<uint8_t>(msg.data[i]);
180 }
181 {
182 InterruptLocker locker;
183 while (CAN_Transmit(periphInfo.splPtr(), &txMsg) == CAN_TxStatus_NoMailBox) {
184 continue;
185 }
186 }
187 return true;
188}

190void CANMethod::interrupt() {
191 if (CAN_GetFlagStatus(periphInfo.splPtr(), CAN_FLAG_FMP0) == RESET) {
192 return;
193 }
194 CanRxMsg rxMsg;
195 CAN_Receive(periphInfo.splPtr(), CAN_FIFO0, &rxMsg);
196
197 peripheral::can::Message msg;
198 if (rxMsg.IDE == CAN_Id_Standard) {
199 msg.idType = peripheral::can::Message::IDType::Standard;
200 msg.id = rxMsg.StdId;
201 } else {
202 msg.idType = peripheral::can::Message::IDType::Extended;
203 msg.id = rxMsg.ExtId;
204 }
205 msg.frameType = rxMsg.RTR == CAN_RTR_Data ? peripheral::can::Message::FrameType::Data :
206 peripheral::can::Message::FrameType::Remote;
207 for (uint32_t i = 0; i < rxMsg.DLC; ++i) {
208 msg.data.push_back(static_cast<std::byte>(rxMsg.Data[i]));
209 }
210 recvCallback(std::move(msg));
211}
