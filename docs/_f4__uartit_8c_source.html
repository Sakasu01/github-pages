94ErrorStatus UARTIT_config(USART_TypeDef* USARTx) {

96 if (!buffer)
97 return ERROR;

150void UARTIT_putch(USART_TypeDef* USARTx, char ch) {
151 UARTIT_BufferTypeDef* buffer = UARTIT_getBuffer(USARTx);
152 if (!buffer)
153 return;
154
155 __disable_irq();
156
157 if (UARTIT_IS_TX_AVAILABLE(buffer)) {
158 buffer->txBuffer[buffer->txWritePt] = (uint8_t)ch;
159 ++buffer->txWritePt;
160 buffer->txWritePt %= UARTIT_BUFSIZE;
161
162 USART_ITConfig(USARTx, USART_IT_TXE, ENABLE);

165 __enable_irq();

168void UARTIT_puts(USART_TypeDef* USARTx, const char* str) {
169 UARTIT_BufferTypeDef* buffer = UARTIT_getBuffer(USARTx);
170 if (!buffer)
171 return;
172
173 __disable_irq();
174
175 while (*str != '\0' && UARTIT_IS_TX_AVAILABLE(buffer)) {
176 buffer->txBuffer[buffer->txWritePt] = (uint8_t)*str++;
177 ++buffer->txWritePt;
178 buffer->txWritePt %= UARTIT_BUFSIZE;
179 }
180
181 if (!UARTIT_IS_TX_EMPTY(buffer))
182 USART_ITConfig(USARTx, USART_IT_TXE, ENABLE);
183
184 __enable_irq();
185}

204static void UARTIT_interruputHander(USART_TypeDef* USARTx) {
205 if (USART_GetITStatus(USARTx, USART_IT_TXE) != RESET) {
206 UARTIT_BufferTypeDef* buffer = UARTIT_getBuffer(USARTx);
207 if (buffer) {
208 if (UARTIT_IS_TX_EMPTY(buffer)) {
209 USART_ITConfig(USARTx, USART_IT_TXE, DISABLE);
210 } else {
211 USART_SendData(USARTx, buffer->txBuffer[buffer->txReadPt]);
212 ++buffer->txReadPt;
213 buffer->txReadPt %= UARTIT_BUFSIZE;
214 }
215 }
216
217 USART_ClearITPendingBit(USARTx, USART_IT_TXE);
