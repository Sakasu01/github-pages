47inline uint8_t Provider::generateFreeTimeStamp()
48{
49 // タイムスタンプの空きができるまで待機
50 // デッドロック発生するとしたらここ
51 // すでに全てのタイムスタンプが使われている状態で最優先割り込みがここにくるとout
52 while (waitingResponseFlag == 0xFFFF)
53 {
54 continue;
55 }
56 uint16_t availableTimeStampBit;
57 {
58 Misc::MutexLocker locker;
59 // 最小値のタイムスタンプ(bit)を発行する
60 availableTimeStampBit = util::bits::lowestOneBit(~waitingResponseFlag);
61 waitingResponseFlag |= availableTimeStampBit;
62 receivedResponseFlag &= ~availableTimeStampBit;
63 }
64 return util::bits::countBits(availableTimeStampBit - 1);
65}

67inline std::optional<Packet::Payload>
68Provider::sendPacketAndWaitResponse(const NodeIDCode &addr,
69 std::uint8_t commandCode,

74 uint8_t timeStamp = generateFreeTimeStamp();

78 packet.header.timeStamp = timeStamp;

82 packet.payload = payload;
83 sendfunc(packet);
84
85 const uint16_t timeStampBit = 1 << timeStamp;
86
87 util::ElapsedTimer elapsed;
88 volatile bool error = false;
89 while (!(receivedResponseFlag & timeStampBit))
90 {
91 if (timeOutUsec && timeOutUsec < elapsed.getElapsedUsec())

93 error = true;
94 break;
95 }
96 }
97
98 // ここフラグ下ろす順番大事なので
99 receivedResponseFlag &= ~timeStampBit;
100 waitingResponseFlag &= ~timeStampBit;
101 if (error)
102 {
103 return std::nullopt;
104 }
105 return receivedResponse[timeStamp].payload;
106}
